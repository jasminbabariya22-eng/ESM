 risk_head_approval_remark varchar(500),
 risk_head_approval_by integer,

 risk_manager_approved_on timestamp without time zone,
 risk_manager_approval_remark varchar(500),
 risk_manager_approval_by integer,
 created_by INTEGER NOT NULL DEFAULT 1,
 created_on TIMESTAMP WITHOUT TIME ZONE NOT NULL DEFAULT NOW(),
 modified_by INTEGER,

 is_active SMALLINT NOT NULL DEFAULT 0,
 is_deleted SMALLINT NOT NULL DEFAULT 0
